def adicionar_tarefa(): input_tarefa = document.getElementById("tarefa_input") nova_tarefa = input_tarefa.value.strip() if nova_tarefa: tarefas.append(nova_tarefa) input_tarefa.value = "" renderizar_lista() salvar_tarefas() mostrar_mensagem("Tarefa adicionada com sucesso!", "green") def remover_tarefa(event): indice_para_remover = int(event.target.value) tarefa_removida = tarefas.pop(indice_para_remover) salvar_tarefas() renderizar_lista() mostrar_mensagem(f"Tarefa '{tarefa_removida}' removida!", "red") def mostrar_mensagem(texto, cor): mensagem_div = document.getElementById("mensagem") mensagem_div.innerText = texto mensagem_div.style.color = cor mensagem_div.style.display = "block" setTimeout(lambda: mensagem_div.style.display = "none", 2000)
